merge_objects

Merge data

Merge R objects of the FGCZ preprocessed BD Rhapsody data. Sample were collected in different experiments and sequencing runs. Here we combine all sample by tumor type and/or condition.

“Old” sample are downloaded from the ScSeurat_2023-12-15--17-47-24 folder and consist of the pooled sample of the first 4 experiments (32384/33286, 32674/33286, 32554/33286). “New” sample were downloaded from the o33260_o33118_ScSeurat_2023-12-01--19-43-37 folder and completely sequenced in a second round and refer to experiment 5 and 6 (33260,33118).

The following objects are generated: sce_tum: All cells from tumor sample from experiments 33260 and 33118 sce_ln: All cells from ln sample from experiments 33260 and 33118 sce_tum_old: All cells from tumor sample from the pooled experiments 32384/33286, 32674/33286 sce_ln_old: All cells from ln sample from the pooled experiments 32384/33286, 32674/33286 sce_tum_yumm: All cells from tumor samples of YUMM type in all experiments sce_tum_yummer: All cells from tumor samples of YUMMER type in all experiments sce_ln_yumm: All cells from LN samples of YUMM type in all experiments sce_ln_yummer: All cells from LN samples of YUMMER type in all experiments

read_seurat <- function(fi_nam){
  seurat <- readRDS(fi_nam)
  seurat$tissue <- gsub("NOV_([^_]*.)(_[^_].*)", "\\1", seurat$Sample)
  seurat$tumor <- gsub("NOV_.*(YUMM|YUMMER|Skin)_([0-9].*)_.*", "\\1", 
                       seurat$Sample)
  seurat$mouse <- gsub("NOV_.*(YUMM|YUMMER|Skin)_([0-9].*)_.*", "\\2", 
                       seurat$Sample)
  seurat$experiment <- "new"
  sce <- as.SingleCellExperiment(seurat)
}

read_seurat_old <- function(fi_nam){
  seurat <- readRDS(fi_nam)
  seurat$tissue <- gsub("^(LN|Tumor)(YUMM|YUMMER|Skin)", "\\1", seurat$Condition)
  seurat$tumor <- gsub("^(LN|Tumor)(YUMM|YUMMER|Skin)", "\\2", seurat$Condition)
  seurat$mouse <- gsub("(Ln|Tumor)(YUMM|YUMMER|Skin)([0-9].*)", "\\3", seurat$Sample)
  seurat$experiment <- "old"
  sce <- as.SingleCellExperiment(seurat)
}

Combine experiments (no batch correction)

merge_sce <- function(sce_list){
  # get common genes
  gene_list <- lapply(sce_list, function(sce){
  rownames(sce)
  })
  universe <- Reduce(intersect, gene_list)
  sce_list <- lapply(sce_list, function(sce){
  sce <- sce[universe,]
  })
  gc()
  # rescale sample
  sce_list <- multiBatchNorm(sce_list)
  # get highly variable genes
  dec_list <- lapply(sce_list, function(sce){
  dec <- modelGeneVar(sce)
  })
  combined.dec <- combineVar(dec_list)
  chosen.hvgs <- combined.dec$bio > 0
  sce <- do.call(cbind, sce_list)
  sce <- runPCA(sce, subset_row=chosen.hvgs,
    BSPARAM=BiocSingular::RandomParam())
  sce <- runTSNE(sce, dimred="PCA")
  sce <- runUMAP(sce, dimred="PCA")
  sce
}
